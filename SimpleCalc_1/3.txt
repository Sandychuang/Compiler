function max(a,b){
 if(a>b) return 1
 else if(a==b) return 0;
 else return 2
}
function fib(a){
 if(a<=0) return 0
 else if(a==1) return 1
 else(a>1) return fib(a-1)+fib(a-2)
}
function fact(a){
 if(a<=0)return 0
 else{
  b=a
  while(a>0){
   b=b*(a-1)   
  }
 }
}
