//symbols
stmt_list stmt 
exp add_exp mul_exp pri_exp num_exp

//tokens
number = + - * / e ( ) eof ;

//grammars
stmt_list -> 
stmt_list stmt | stmt

stmt ->
exp ;

exp ->
exp + add_exp | exp - add_exp | add_exp

add_exp ->
add_exp + mul_exp | add_exp - mul_exp | mul_exp

mul_exp -> 
mul_exp * pri_exp | mul_exp / pri_exp | pri_exp

pri_exp -> 
num_exp | ( exp )

num_exp -> 
number 

//end
